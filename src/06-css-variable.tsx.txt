import React from 'react';
import './App.css';
// react-css-variable
// setInterval, counter, random sort, start/stop, isRunning
// camel names


const { useState, useEffect } = React;

function App() {
  const [counter, setCounter] = useState(0);
  const [isRunning, setIsRunning] = useState(false);
  colorlist.sort(() => Math.random() - 0.5);
  const [bgcolor, setBGColor] = useState(colorlist[0]);
  const delay: number = 2000;

  document.documentElement.style.setProperty("--background", bgcolor);

  useEffect(() => {
    const interval = setInterval(() => {
      if (isRunning) {
        setCounter((prevCount) => (prevCount + 1) % colorlist.length);
        setBGColor(colorlist[counter]);
      }
    }, delay);
    return () => {
      clearInterval(interval);
    };
  });

  return (
    <div className="App">
      <h1>{bgcolor}</h1>
      <button onClick={() => setIsRunning(true)}>START</button>
      <button
        onClick={() => {
          setIsRunning(false);
          console.log(counter);
        }}
      >
        STOP
      </button>
    </div>
  );
}
